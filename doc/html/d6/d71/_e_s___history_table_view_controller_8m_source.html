103 - (void)recalculateEventsFromPredictionList
104 {
105  // Empty the event history:
106  [self.eventHistory removeAllObjects];
107  ES_AppDelegate *appDelegate = [[UIApplication sharedApplication] delegate];
108 
109  NSMutableArray *predictions = appDelegate.predictions;
110  // Read the prediction list of the user and group together consecutive timepoints with similar activities to unified activity events:
111  ES_Activity *startOfActivity = nil;
112  ES_Activity *endOfActivity = nil;
113  ES_ActivityEvent *currentEvent = nil;
114  NSMutableArray *minuteActivities = nil;
115  for (id activityObject in [predictions reverseObjectEnumerator])
116  {
117  ES_Activity *currentActivity = (ES_Activity *)activityObject;
118 
119  if (![[self class] isActivity:currentActivity similarToActivity:startOfActivity])
120  {
121  // Then we've reached a new activity.
122  if (startOfActivity)
123  {
124  // Create an event from the start and end of the previous activity:
125  currentEvent = [[ES_ActivityEvent alloc] initWithIsVerified:startOfActivity.isPredictionVerified serverPrediction:startOfActivity.serverPrediction userCorrection:startOfActivity.userCorrection userActivityLabels:startOfActivity.userActivityLabels startTimestamp:startOfActivity.timestamp endTimestamp:endOfActivity.timestamp minuteActivities:minuteActivities];
126  [self.eventHistory addObject:currentEvent];
127  }
128 
129  //update the new "start" for the current activity:
130  minuteActivities = [[NSMutableArray alloc] initWithCapacity:1];
131  startOfActivity = currentActivity;
132  }
133  // Add teh minute activity object to the list of minutes of the current event:
134  [minuteActivities addObject:currentActivity];
135  endOfActivity = currentActivity;
136  }
137  if (endOfActivity)
138  {
139  // Create the last event from the start and end of activity:
140  ES_ActivityEvent *event = [[ES_ActivityEvent alloc] initWithIsVerified:startOfActivity.isPredictionVerified serverPrediction:startOfActivity.serverPrediction userCorrection:startOfActivity.userCorrection userActivityLabels:startOfActivity.userActivityLabels startTimestamp:startOfActivity.timestamp endTimestamp:endOfActivity.timestamp minuteActivities:minuteActivities];
141  [self.eventHistory addObject:event];
142  }

144 }
